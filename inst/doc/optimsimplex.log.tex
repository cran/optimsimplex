\HeaderA{optimsimplex.log}{Optimsimplex Logging}{optimsimplex.log}
\keyword{method}{optimsimplex.log}
%
\begin{Description}\relax
This function prints a message to screen (or log file).
\end{Description}
%
\begin{Usage}
\begin{verbatim}
  optimsimplex.log(this = NULL, msg = NULL)
\end{verbatim}
\end{Usage}
%
\begin{Arguments}
\begin{ldescription}
\item[\code{this}] An simplex object.
\item[\code{msg}] A message to print.
\end{ldescription}
\end{Arguments}
%
\begin{Value}
Do not return any value but print \code{msg} to screen if the \code{verbose}
in \code{this} is set to 1.
\end{Value}
%
